🔗 IMPORTANT ROUTES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  PUBLIC ROUTES:
    GET  /blog/               → View all blog posts
    GET  /blog/<slug>/        → View individual blog post

  DASHBOARD ROUTES:
    GET  /dashboard/blog/               → List all posts
    GET  /dashboard/blog/add/           → Add new post form
    POST /dashboard/blog/add/           → Create post
    GET  /dashboard/blog/<id>/edit/     → Edit post form
    POST /dashboard/blog/<id>/edit/     → Update post
    POST /dashboard/blog/<id>/delete/   → Delete post

    GET  /dashboard/site-images/       → List all images
    GET  /dashboard/site-images/add/   → Upload image form
    POST /dashboard/site-images/add/   → Upload image
    GET  /dashboard/site-images/<id>/edit/   → Edit image
    POST /dashboard/site-images/<id>/edit/   → Update image
    POST /dashboard/site-images/<id>/delete/ → Delete image

  ADMIN ROUTES:
    /admin/blog/blogpost/      → Blog admin
    /admin/dashboard/siteimage/ → Image admin
